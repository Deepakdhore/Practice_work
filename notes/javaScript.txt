	number	All numbers (integers, floats)			let age: number = 25;
	string	Text values					let name: string = "Deepak";
	boolean	true or false					let isOnline: boolean = true;
	any	Can be anything (turns off type safety)		let data: any = 42;
	unknown	Like any, but saferâ€”you must check type first	let value: unknown = "hello";
	void	Used for functions that return nothing		function log(): void { console.log("Hi"); }
