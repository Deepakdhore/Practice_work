interface PostData {
  title: string;
  body: string;
  userId: number;
}

async function createPost(post: PostData): Promise<void> {
  const response = await fetch('https://jsonplaceholder.typicode.com/posts', {
   
    method: "POST",                     -----1
    headers: {                          ------2
      "Content-Type": "application/json"
    },
    body: JSON.stringify(post)          ------3
  
  });

  const result = await response.json();
  console.log("Created Post:", result);
}

createPost({
  title: "My new post",
  body: "This is the body",
  userId: 1
});
